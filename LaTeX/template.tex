\newcommand{\get}[1]{\pgfkeysvalueof{/TemplateVersion0/#1}}
\newcommand{\seperator}{\par\noindent\rule{\textwidth}{0.4pt}}
\newcommand{\seperateAuthors}[1]{\noindent\StrSubstitute{\get{properties/Authors}}{&&}{#1}}
