\RequirePackage{xstring}


% Check definitions
\makeatletter
\@ifundefined{templateVersion}{
	\errmessage{Choose a template version! You can use "latest" for the latest version}
	\stop
}
\@ifundefined{templatePath}{
	\ClassWarningNoLine{Template}{You did not set the template path. I will use the current directory}
	\def\templatePath{.}
}
\makeatother


% Check choosen version
\IfStrEqCase{\templateVersion}{
	{latest}{\def\templateVersion{1}}
	{1}{}
}[\errmessage{Unknown version: \templateVersion}\stop]


% Load choosen version
\newcommand{\versionPath}{\templatePath/Version/\templateVersion}
\makeatletter
\def\input@path{{\versionPath}}
\makeatother
\input{setup.tex}
